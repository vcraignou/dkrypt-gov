  <script>
    const URL = "https://vcraignou.github.io/dkrypt-gov/snapshots.json?v=" + Date.now();

    function fmtNumber(x) {
      if (x === null || x === undefined || x === "") return "n/a";
      const n = Number(x);
      if (Number.isNaN(n)) return x;

      if (Math.abs(n) >= 1_000_000_000) return (n / 1_000_000_000).toFixed(2) + "B";
      if (Math.abs(n) >= 1_000_000) return (n / 1_000_000).toFixed(2) + "M";
      if (Math.abs(n) >= 1_000) return (n / 1_000).toFixed(2) + "K";

      return n.toFixed(2);
    }

    function fmtPct(x) {
      if (x === null || x === undefined || x === "") return "n/a";
      const n = Number(x);
      if (Number.isNaN(n)) return x;
      return n.toFixed(1) + "%";
    }

    function renderPeerTable(peerTable) {
      if (!peerTable || peerTable.length === 0) {
        return "<div class='text-block'>No peer table available.</div>";
      }

      let html = "<table><thead><tr><th>Name</th><th>FDV / Fees</th><th>FDV / Revenues</th></tr></thead><tbody>";

      peerTable.forEach(row => {
        html += `
          <tr class="${row.highlight ? "highlight" : ""}">
            <td>${row.name ?? "n/a"}</td>
            <td>${fmtNumber(row.fdv_fees_90d)}</td>
            <td>${fmtNumber(row.fdv_revenues_90d)}</td>
          </tr>
        `;
      });

      html += "</tbody></table>";
      return html;
    }

    async function load() {
      try {
        const res = await fetch(URL, { cache: "no-store" });
        if (!res.ok) throw new Error("HTTP " + res.status);

        const data = await res.json();

        document.getElementById("generated").innerText =
          `Generated: ${data.generated_at || "n/a"} | Tokens: ${data.count || 0}`;

        const container = document.getElementById("snapshots");
        container.innerHTML = "";

        (data.snapshots || []).forEach(s => {
          const govScore = s.governance?.score || "n/a";

          const div = document.createElement("div");
          div.className = "snapshot";

          div.innerHTML = `
            <div class="token-header">
              <div>
                <div class="token">${s.token} — ${s.protocol}</div>
                <div class="badges">
                  <span class="badge ${govScore}">Governance: ${String(govScore).toUpperCase()}</span>
                  <span class="badge">Accrual: ${s.governance?.revenue_accrual_type || "n/a"}</span>
                  <span class="badge">FDV / Fees: ${fmtNumber(s.valuation?.fdv_fees_90d)}x</span>
                  <span class="badge">12m unlock: ${fmtPct(s.tokenomics?.unlock_12m_pct)}</span>
                </div>
              </div>
            </div>

            <div class="section-title">Investment conclusion</div>
            <div class="conclusion">${s.conclusion || "n/a"}</div>

            <div class="section-title">Key metrics</div>
            <div class="grid">
              <div class="metric">
                <div class="metric-label">Market cap</div>
                <div class="metric-value">${fmtNumber(s.tokenomics?.market_cap)}</div>
              </div>
              <div class="metric">
                <div class="metric-label">FDV</div>
                <div class="metric-value">${fmtNumber(s.tokenomics?.fdv)}</div>
              </div>
              <div class="metric">
                <div class="metric-label">MC / FDV</div>
                <div class="metric-value">${fmtNumber(s.tokenomics?.mc_fdv)}</div>
              </div>
              <div class="metric">
                <div class="metric-label">Unlocked</div>
                <div class="metric-value">${fmtPct(s.tokenomics?.unlocked_pct)}</div>
              </div>
              <div class="metric">
                <div class="metric-label">Fees 90d ARR</div>
                <div class="metric-value">${fmtNumber(s.financials?.fees_90d_arr)}</div>
              </div>
              <div class="metric">
                <div class="metric-label">Revenue 90d ARR</div>
                <div class="metric-value">${fmtNumber(s.financials?.revenues_90d_arr)}</div>
              </div>
            </div>

            <div class="section-title">Overview</div>
            <div class="text-block">${s.overview || "n/a"}</div>

            <div class="section-title">Team</div>
            <div class="text-block">${s.team || "n/a"}</div>

            <div class="section-title">Product / market fit</div>
            <div class="text-block">${s.product_market_fit || "n/a"}</div>

            <div class="section-title">Governance and value accrual</div>
            <div class="text-block">
              Dual Labs-DAO structure: <b>${s.governance?.dual_lab_dao_structure}</b><br>
              DAO decentralised enough: <b>${s.governance?.dao_decentralised_enough}</b><br>
              Direct revenue accrual: <b>${s.governance?.direct_revenue_accrual_to_protocol}</b><br>
              Notes: ${s.governance?.notes || "n/a"}
            </div>

            <div class="section-title">Relative valuation</div>
            ${renderPeerTable(s.valuation?.peer_table)}
          `;

          container.appendChild(div);
        });

      } catch (err) {
        document.getElementById("generated").innerText =
          "Could not load token snapshots: " + err.message;
      }
    }

    load();
  </script>
